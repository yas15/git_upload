Chaining Makes It Easier
Without Chaining: Re-selection (avoid this approach)
$('#special-features li').width('50%');
$('#special-features li').height('200px')
$('#special-features li').addClass('highlight bordered');
Without Chaining: Temp Variable
var features = $('#special-features li');
features.width('50%');
features.height('200px')
features.addClass('highlight bordered');
Sets (no variable)
$('#special-features li')
    .width('50%')
    .height('200px')
    .addClass('highlight bordered') 
    .each(function(index, item){
      var $item = $(item);
      $item.text($item.text() + ' ' + 
               $item.attr('data-feature-id'));
    });
Sets (with a variable)
var items = $('#special-features li');
  
items.width('50%')
     .height('200px')
     .addClass('highlight bordered')
     .each(function(index, item){
          var $item = $(item);
          $item.text($item.text() + ' ' + 
             $item.attr('data-feature-id'));
     });
